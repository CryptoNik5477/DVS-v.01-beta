    const niches = [

      "Crypto & Blockchain",

      "Gaming & Esport",

      "Intelligence Artificielle",

      "Memes Internet",

      "Politique & Actualités",

      "Sport & Événements",

      "NFT & Art Digital",

      "DeFi & Finance",

      "Musique & Célébrités",

      "Cinéma & Séries"

    ];

    const fakeData = {

      "Crypto & Blockchain": ["Bitcoin ETF", "Layer 2 wars", "Memecoin season"],

      "Gaming & Esport": ["GTA 6 leak", "Valorant Champions", "Elden Ring DLC"],

      "Intelligence Artificielle": ["GPT-5 rumors", "AI music", "Deepfake ban"],

      "Memes Internet": ["Doge comeback", "Wojak NFT", "Pepe the Frog"],

      "Politique & Actualités": ["US elections", "Climate strikes", "Elon tweet"],

      "Sport & Événements": ["World Cup 2026", "Messi Miami", "NBA finals"],

      "NFT & Art Digital": ["Pudgy Penguins", "Ordinals", "AI art hype"],

      "DeFi & Finance": ["Yield farming 2.0", "RWA token", "ZK-rollups"],

      "Musique & Célébrités": ["Taylor Swift album", "Drake AI", "Coachella"],

      "Cinéma & Séries": ["Dune 3", "Stranger Things", "Marvel reboot"]

    };

    function populateAgents() {

      const grid = document.getElementById('agentGrid');

      niches.forEach(niche => {

        const card = document.createElement('div');

        card.className = 'card';

        const trends = fakeData[niche] || ["Trend 1", "Trend 2"];

        const selected = trends.sort(() => 0.5 - Math.random()).slice(0, 2);

        const noise = ["🔥","🚀","💎","📈",""];

        const display = selected.map(t => t + ' ' + noise[Math.floor(Math.random() * noise.length)]).join(', ');

        card.innerHTML = `<strong>${niche}</strong><div class="trends">${display}</div>`;

        grid.appendChild(card);

      });

    }

    function getReports() {

      let reports = [];

      niches.forEach(niche => {

        const trends = fakeData[niche] || ["Trend 1"];

        const selected = trends.sort(() => 0.5 - Math.random()).slice(0, 2);

        reports.push(`[${niche}] ${selected.join(', ')}`);

      });

      return reports.join('\n');

    }

    async function synthetise() {

      const apiKey = document.getElementById('apiKey').value.trim();

      if (!apiKey.startsWith('sk-')) {

        alert('Veuillez entrer une clé API OpenAI valide (commence par sk-...).');

        return;

      }

      const reports = getReports();

      const prompt = `Tu es un expert en meme coins. Voici les rapports de tendances de 10 niches différentes :\n\n${reports}\n\nSélectionne les 3 meilleures idées de meme coins qui pourraient percer MAINTENANT. Pour chaque idée, propose :\n- Un ticker (3 à 5 lettres, ex: $PEPE)\n- Un nom de coin accrocheur\n- Une description courte (1 phrase)\nRéponds UNIQUEMENT avec un JSON valide comme ceci :\n[\n  {\"ticker\": \"PEPE\", \"name\": \"Pepe Coin\", \"description\": \"Le retour du roi des memes\"},\n  ...\n]`;

      document.getElementById('top3Section').style.display = 'none';

      const container = document.getElementById('top3Container');

      container.innerHTML = '<p style="text-align:center">⏳ Analyse en cours...</p>';

      try {

        const response = await fetch('https://api.openai.com/v1/chat/completions', {

          method: 'POST',

          headers: {

            'Content-Type': 'application/json',

            'Authorization': `Bearer ${apiKey}`

          },

          body: JSON.stringify({

            model: 'gpt-3.5-turbo',

            messages: [{ role: 'user', content: prompt }],

            temperature: 0.9

          })

        });

        const data = await response.json();

        if (data.error) {

          throw new Error(data.error.message);

        }

        const content = data.choices[0].message.content.trim();

        const jsonStr = content.replace(/```json|```/g, '').trim();

        const coins = JSON.parse(jsonStr);

        displayTop3(coins, apiKey);

      } catch (error) {

        console.error(error);

        alert('Erreur lors de la synthèse : ' + error.message);

        displayTop3([

          { ticker: 'PEPE', name: 'Pepe Coin', description: 'Le retour du roi des memes' },

          { ticker: 'WOJAK', name: 'Wojak Finance', description: 'La finance du peuple triste' },

          { ticker: 'DOGE', name: 'Doge 2.0', description: 'Le chien qui valait 1 milliard' }

        ], null);

      }

    }

    async function displayTop3(coins, apiKey) {

      const container = document.getElementById('top3Container');

      container.innerHTML = '';

      const section = document.getElementById('top3Section');

      section.style.display = 'block';

      for (const coin of coins) {

        const coinDiv = document.createElement('div');

        coinDiv.className = 'coin';

        const ticker = coin.ticker.toUpperCase();

        let logoHtml = '';

        if (apiKey) {

          try {

            const imgResp = await fetch('https://api.openai.com/v1/images/generations', {

              method: 'POST',

              headers: {

                'Content-Type': 'application/json',

                'Authorization': `Bearer ${apiKey}`

              },

              body: JSON.stringify({

                prompt: `A cute, meme-style logo for a cryptocurrency called ${ticker}, digital art`,

                n: 1,

                size: '256x256'

              })

            });

            const imgData = await imgResp.json();

            if (imgData.data && imgData.data.length > 0) {

              logoHtml = `<img src="${imgData.data[0].url}" alt="Logo ${ticker}">`;

            } else {

              throw new Error('Pas d image');

            }

          } catch (e) {

            logoHtml = `<div class="emoji-logo">${getEmoji(ticker)}</div>`;

          }

        } else {

          logoHtml = `<div class="emoji-logo">${getEmoji(ticker)}</div>`;

        }

        coinDiv.innerHTML = `

          ${logoHtml}

          <div class="ticker">$${ticker}</div>

          <strong>${coin.name}</strong>

          <p style="font-size:0.8em; color:#aaa;">${coin.description}</p>

        `;

        container.appendChild(coinDiv);

      }

    }

    function getEmoji(ticker) {

      const map = {

        'PEPE': '🐸',

        'WOJAK': '😢',

        'DOGE': '🐕',

        'MOON': '🌕',

        'LAMBO': '🏎️',

        'CHAD': '💪',

        'BONK': '🐶',

        'SHIB': '🐕',

      };

      return map[ticker] || '🪙';

    }
